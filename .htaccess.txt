<IfModule mod_rewrite.c>
    RewriteEngine On
    
    # Si el archivo o carpeta existe físicamente (ej. imágenes, css), lo muestra
    RewriteCond %{REQUEST_FILENAME} !-f
    RewriteCond %{REQUEST_FILENAME} !-d
    
    # Si no existe, le pasa la ruta a CodeIgniter (al index.php)
    RewriteRule ^(.*)$ index.php/$1 [L,QSA]
</IfModule>
